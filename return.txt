    private void rentActionPerformed(java.awt.event.ActionEvent evt) {                                     
        try {
            // TODO add your handling code here:
            Class.forName("com.mysql.jdbc.Driver");
            Connection con;

            con = DriverManager.getConnection("jdbc:mysql://localhost:3306/user_database","root","");
            

            SimpleDateFormat df=new SimpleDateFormat("yyy-MM-dd");
            String date=df.format(return_date.getDate());
            String carid=idcar.getText();
            
            PreparedStatement ps2;
            ps2 = con.prepareStatement("select DATEDIFF(NOW(),booking_date) as elap from cust_order where order_id=?");
            ps2.setString(1,orderid.getText());
              ResultSet rs2=ps2.executeQuery();
             rs2.next();
              String elp=rs2.getString("elap");
             int elaped =Integer.parseInt(elp);
            
            String query1="DELETE FROM `cust_order` WHERE `order_id`=?";
            PreparedStatement ps1=con.prepareStatement(query1);
             ps1.setString(1,orderid.getText());
           
             //to retreieve cost of car
               PreparedStatement ps3;
            ps3 = con.prepareStatement("SELECT `car_cost` FROM `car` WHERE `car_id`=?");
            ps3.setString(1,carid);
              ResultSet rs3=ps3.executeQuery();
             rs3.next();
              String cost=rs3.getString("car_cost"); 
               int costi =Integer.parseInt(cost);
             
             int trent=elaped*costi;
             String ren=String.valueOf(trent);
             PreparedStatement ps4;
            ps4 = con.prepareStatement("INSERT INTO `bills`(`oder_id`, `car_id`, `days`, `amount`, `cust_id`,`emp_id`) VALUES (?,?,?,?,?,?)");
            ps4.setString(1,orderid.getText());
            ps4.setString(2,carid) ;
            ps4.setString(3,elp);
             ps4.setString(4,ren);
             ps4.setString(5,custid.getText());
            ps4.setString(6,emp.getText());
             
            if(ps1.executeUpdate()>0)
            {
                JOptionPane.showMessageDialog(null,"calculated rent for "+elp+" days is "+trent);
                
                showtable();
                
                
            }

        } catch (ClassNotFoundException ex) {
